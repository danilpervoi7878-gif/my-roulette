import asyncio import logging import time from aiogram import Bot, Dispatcher, types, F from aiogram.types import ReplyKeyboardMarkup, KeyboardButton, WebAppInfo, InlineKeyboardMarkup, InlineKeyboardButton from aiogram.filters import Command # ТОКЕН И НИК АДМИНА API_TOKEN = 'ТВОЙ_ТОКЕН_БОТА' ADMIN_USERNAME = 'ТВОЙ_НИК' # Без значка @ logging.basicConfig(level=logging.INFO) bot = Bot(token=API_TOKEN) dp = Dispatcher() user_balances = {} prize_names = { "🖼️":"Фото", "🎥":"Видео", "🏆":"Пак 8+2", "🎰":"Пак 10+4", "💄":"Секс", "🍑":"Анал", "🔥":"Минет", "🍆":"Игрушка", "🔑":"Приватка", "☎️":"Звонок", "❤️":"Вирт" } def get_keyboard(user_id): balance = user_balances.get(user_id, 250) # Ссылка на твой GitHub Pages web_app_url = f"https://danilpervoi7878-gif.github.io/my-roulette/play.html?v={int(time.time())}&balance={balance}" return ReplyKeyboardMarkup( keyboard=[[KeyboardButton(text="ИГРАТЬ 🏆", web_app=WebAppInfo(url=web_app_url))]], resize_keyboard=True ) @dp.message(Command("start")) async def send_welcome(message: types.Message): user_id = message.from_user.id if user_id not in user_balances: user_balances[user_id] = 250 await message.answer( f"Привет! 👋\nТвой баланс: ⭐ {user_balances[user_id]}\n\nЖми кнопку ниже, чтобы начать игру!", reply_markup=get_keyboard(user_id) ) @dp.message(F.web_app_data) async def get_web_app_data(message: types.Message): user_id = message.from_user.id try: data = message.web_app_data.data.split('|') price = int(data[0]) prize_emoji = data[1] if user_id in user_balances: user_balances[user_id] -= price if user_balances[user_id] < 0: user_balances[user_id] = 0 prize_text = prize_names.get(prize_emoji, "Приз") kb_contact = InlineKeyboardMarkup(inline_keyboard=[ [InlineKeyboardButton(text="ЗАБРАТЬ ПРИЗ 🎁", url=f"https://t.me/{ADMIN_USERNAME}")] ]) await message.answer( f"✅ **Выигрыш зафиксирован!**\n\n" f"Твой приз: {prize_emoji} **{prize_text}**\n" f"Списано: {price} ⭐\n" f"Остаток баланса: {user_balances[user_id]} ⭐\n\n" "Нажми кнопку ниже, чтобы забрать награду!", reply_markup=kb_contact, parse_mode="Markdown" ) await message.answer( "Хочешь испытать удачу еще раз? 👇", reply_markup=get_keyboard(user_id) ) except Exception as e: logging.error(f"Ошибка: {e}") async def main(): await dp.start_polling(bot) if __name__ == '__main__': asyncio.run(main())
